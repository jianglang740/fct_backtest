# 单因子回测流程图

```mermaid
flowchart TB
    %% ==================== 样式定义 ====================
    classDef dataFile fill:#1a1a2e,stroke:#16213e,color:#e0e0e0
    classDef process fill:#0f3460,stroke:#16213e,color:#e0e0e0
    classDef filter fill:#533483,stroke:#3b2770,color:#e0e0e0
    classDef analysis fill:#1a3a3a,stroke:#0f2f2f,color:#e0e0e0
    classDef metrics fill:#4a2e1e,stroke:#3a1e0e,color:#e0e0e0
    classDef output fill:#2e1e4a,stroke:#1e0e3a,color:#e0e0e0
    classDef decision fill:#5c3d2e,stroke:#3d2e1e,color:#e0e0e0

    %% ==================== 数据加载 ====================
    subgraph S1["📁 第一步：数据加载"]
        direction LR
        A1["📦 price_data.parquet<br/>577万行 × 13列<br/>MultiIndex: [trade_date, code]<br/>──────────────<br/>is_st · trade_days<br/>open/high/low/close<br/>vol · amount · pct_chg<br/>open_adj · high_adj<br/>low_adj · close_adj"]:::dataFile
        A2["📦 fct_df.parquet<br/>566万行 × 2列<br/>MultiIndex: [trade_date, code]<br/>──────────────<br/>fct_1 · fct_2"]:::dataFile
    end

    %% ==================== 数据合并 ====================
    subgraph S2["🔗 第二步：数据合并"]
        B1["pd.concat(axis=1, join='inner')<br/>按 MultiIndex [trade_date, code] 水平拼接<br/>取交集：仅保留两个文件共有的行<br/>──────────────<br/>合并后: 566万行 × 15列"]:::process
    end

    %% ==================== 因子预处理 ====================
    subgraph S3["🔧 第三步：因子预处理"]
        direction TB
        C1["extreme_MAD()<br/>──────<br/>稳健缩尾去极值<br/>上下界 = 中位数 ± 5.2×MAD<br/>低于下界→截断为下界<br/>高于上界→截断为上界"]:::process
        C2["standardize_z() + fillna(0)<br/>──────<br/>Z-Score 标准化<br/>(x − mean) / std<br/>缺失值填充为 0（均值水平）"]:::process
        C1 --> C2
    end

    %% ==================== 样本池过滤 ====================
    subgraph S4["🔍 第四步：样本池过滤 (preprocess_data)"]
        direction TB
        D1{"is_st == False ?<br/>──────<br/>剔除 ST 股票<br/>（约 17.7万行，占 3%）"}:::filter
        D2{"trade_days ≥ 365 ?<br/>──────<br/>剔除上市不足一年<br/>（约 34万行，占 6%）"}:::filter
        D3["删除 is_st 和 trade_days 列<br/>──────<br/>精简数据，后续不再使用"]:::filter
        D1 -->|✅ 是| D2
        D1 -->|❌ 否 ST| D_OUT["🚫 丢弃"]
        D2 -->|✅ 是| D3
        D2 -->|❌ 否 次新股| D_OUT
    end

    %% ==================== 因子回测核心 ====================
    subgraph S5["⚙️ 第五步：因子分层回测 (factor_analysis)"]
        direction TB
        E1["构建 close_adj 透视表<br/>──────<br/>pivot_table(index='trade_date',<br/>columns='code', values='close_adj')<br/>变为 trade_date × code 二维矩阵"]:::process

        E2["对每个持有期 period ∈ {1, 5, 10, 20}<br/>循环执行以下步骤"]:::process

        subgraph S5_LOOP["单期回测循环"]
            direction TB
            E3["计算未来收益率<br/>──────<br/>future = close_adj.shift(-period-1)  → 卖出价<br/>current = close_adj.shift(-1)          → 买入价<br/>收益率 = future / current − 1"]:::process
            E4["避免样本重叠<br/>──────<br/>iloc[::period] 每隔 period 行取一日期<br/>使收益序列独立同分布<br/>便于 ICIR 统计推断"]:::process
            E5["因子-收益对齐<br/>──────<br/>pd.merge(factor_df, returns_series)<br/>按 (trade_date, code) 合并<br/>得到: 日期 · 代码 · 因子值 · 未来收益"]:::process
            E6["每日截面分层<br/>──────<br/>pd.qcut(x, 10, duplicates='drop')<br/>每天按因子值分 10 组<br/>Q1(最低) ~ Q10(最高)<br/>加入 quantile 列标记所属分组"]:::process
            E3 --> E4 --> E5 --> E6
        end

        E2 --> E3
    end

    %% ==================== 绩效分析 ====================
    subgraph S6["📊 第六步：绩效分析 (analyze_factor_performance)"]
        direction TB
        F1["计算分位数日收益率<br/>──────<br/>groupby(date, quantile).mean()<br/>透视表: trade_date × quantile"]:::analysis
        F2["计算分位数累计收益<br/>──────<br/>(1+r).cumprod() − 1<br/>取最后一行 = 全期累计收益率"]:::analysis
        F3["计算多空组合收益<br/>──────<br/>Q10 累计收益 − Q1 累计收益"]:::analysis
        F4["计算 IC 时间序列<br/>──────<br/>每日 Spearman 秩相关系数<br/>corr(因子值, 未来收益)<br/>样本 ≤5 则丢弃该日"]:::analysis
        F5["计算 ICIR<br/>──────<br/>IC 均值 / IC 标准差<br/>衡量因子预测稳定性"]:::analysis
        F1 --> F2
        F2 --> F3
        F1 --> F4
        F4 --> F5
    end

    %% ==================== 方向判断 ====================
    subgraph S7["🧭 第七步：因子方向判断 (determine_factor_direction)"]
        direction TB
        G1{"多空收益 > 0 ?"}:::decision
        G2["✅ 正向因子<br/>因子值越大 → 收益越高<br/>目标分组: Top组 Q10"]:::analysis
        G3["🔄 负向因子<br/>因子值越小 → 收益越高<br/>目标分组: Bottom组 Q1"]:::analysis
        G1 -->|是| G2
        G1 -->|否| G3
    end

    %% ==================== 组合指标 ====================
    subgraph S8["📈 第八步：组合绩效指标 (calculate_portfolio_metrics)"]
        direction TB
        H1["对目标分位组（正向→Q10 / 负向→Q1）的日收益率序列<br/>计算以下指标"]:::metrics
        H2["累计收益率<br/>∏(1+r) − 1"]:::metrics
        H3["年化收益率<br/>(1+R)^(252/n) − 1"]:::metrics
        H4["最大回撤<br/>max(1 − 净值/历史最高)"]:::metrics
        H5["夏普比率<br/>年化收益 / 年化波动率"]:::metrics
        H6["索提诺比率<br/>年化收益 / 下行波动率"]:::metrics
        H7["卡玛比率<br/>年化收益 / 最大回撤"]:::metrics
        H1 --> H2 & H3 & H4 & H5 & H6 & H7
    end

    %% ==================== 输出 ====================
    subgraph S9["📤 第九步：结果输出"]
        direction LR
        I1["📊 Excel 输出<br/>──────<br/>Sheet1: 汇总统计<br/>IC · ICIR · 收益 · 回撤 · 夏普<br/>──────<br/>Sheet2: 分位数收益<br/>Q1~Q10 各期累计收益<br/>──────<br/>Sheet3: IC时间序列<br/>每日IC值"]:::output
        I2["🎨 可视化图表<br/>──────<br/>图1: 分层累计收益曲线<br/>Q1~Q10 各组的净值走势<br/>Q1(蓝) Q10(红) 突出显示<br/>──────<br/>图2: 因子绩效仪表盘<br/>多空收益柱状图<br/>IC柱状图<br/>分位收益线图<br/>IC时间序列图"]:::output
    end

    %% ==================== 主流程连线 ====================
    S1 --> S2
    S2 --> S3
    S3 --> S4
    S4 --> S5
    S5 --> S6
    S6 --> S7
    S7 --> S8
    S8 --> S9

    %% ==================== 图例 ====================
    subgraph LEG["图例"]
        L1["📦 数据文件"]:::dataFile
        L2["🔧 数据处理"]:::process
        L3["🔍 过滤筛选"]:::filter
        L4["⚙️ 回测分析"]:::analysis
        L5["📈 绩效指标"]:::metrics
        L6["📤 结果输出"]:::output
    end
```
